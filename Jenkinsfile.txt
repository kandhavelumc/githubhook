pipeline {
    agent any

    stages {
        stage('Build') {
            steps {
                echo 'Building..'
            }
        }
        stage('Test') {
            steps {
                echo 'Testing..'
            }
        }
        stage('Deploy') {
            steps {
                echo 'Deploying....'
            }
        }
    }

        always {
            echo 'I will do this 
                  no matter what the status is'
        }

            message: 'OK')
        }

            message: 'FAIL')
        }
    }
